1function [Qcq,Qcp,Qcc]= CoreControl(k_cc,p_cc,r_cc,n,K_t0,K_q0,D,Tr,rho,eps_c,K_t0r,K_q0r)
 2%#codegen
 3
 4lambda_c = 0.5+0.5*tanh(eps_c*n/5);
 5K_TC = K_t0*lambda_c+(1-lambda_c)*K_t0r;
 6K_QC = K_q0*lambda_c+(1-lambda_c)*K_q0r;
 7Qcq = (K_QC/K_TC)*D*Tr;
 8if n == 0
 9    Qcp = 0;
10else
11Qcp = sign(Tr)*K_QC*(abs(Tr)^(3/2))/(sqrt(rho)*D*K_TC^(3/2)*abs(n));
12end
13alpha_c = exp(-k_cc*abs(p_cc*n)^r_cc);
14
15Qcc = alpha_c*Qcq + (1-alpha_c)*Qcp;
